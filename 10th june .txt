// Add Shards
sh.addShard("shard1ReplSet/localhost:27011");
sh.addShard("shard2ReplSet/localhost:27012");

// Enable Sharding
sh.enableSharding("eCommerceDB");

// Shard Collection using Hashed Shard Key
sh.shardCollection(
    "eCommerceDB.orders",
    { customer_id: "hashed" }
);

// Switch Database
db = db.getSiblingDB("eCommerceDB");

// Insert Sample Data
for (let i = 0; i < 1000; i++) {
    db.orders.insertOne({
        customer_id: "cust_" + i,
        amount: Math.floor(Math.random() * 1000),
        status: "Pending"
    });
}

// Verify Data
print("Total Documents:");
print(db.orders.countDocuments());

// Display Shard Distribution
print("\nShard Distribution:");
db.orders.getShardDistribution();

// Display Cluster Status
print("\nCluster Status:");
sh.status();
