1. Employee Name, Department Name, and Location Name

ANSWER
SELECT e.employee_name,
              d.dept_name,
              d.location_id
FROM Employees e
JOIN Departments d
ON e.dept_id = d.dept_id;

2. Project Name, Client Name, and Total Hours Worked for Consulting Department

ANSWER
SELECT p.project_name,
              p.client_name,
              SUM(a.hours_worked) AS total_hours
FROM Projects p
JOIN Assignments a
ON p.project_id = a.project_id
JOIN Departments d
ON p.dept_id = d.dept_id
WHERE d.dept_name = 'Consulting'
GROUP BY p.project_name, p.client_name;

3. Employees who worked on projects for 'GlobalTech Solutions'

ANSWER
SELECT e.employee_name,
              p.project_name
FROM Employees e
JOIN Assignments a
ON e.employee_id = a.employee_id
JOIN Projects p
ON a.project_id = p.project_id
WHERE p.client_name = 'GlobalTech Solutions';

4. Departments with no Employees

ANSWER
SELECT d.dept_name
FROM Departments d
LEFT JOIN Employees e
ON d.dept_id = e.dept_id
WHERE e.employee_id IS NULL;

5. Manager Name, Department Name, and total direct reports

ANSWER
SELECT m.employee_name AS manager_name,
              d.dept_name,
              COUNT(e.employee_id) AS direct_reports
FROM Employees e
JOIN Employees m
ON e.manager_id = m.employee_id
JOIN Departments d
ON m.dept_id = d.dept_id
GROUP BY m.employee_name, d.dept_name;

6. Average Hours Worked per Employee

ANSWER
SELECT e.employee_name,
              AVG(a.hours_worked) AS average_hours
FROM Employees e
JOIN Assignments a
ON e.employee_id = a.employee_id
GROUP BY e.employee_name;

7. Departments with average salary above company average

ANSWER
SELECT d.dept_name,
              AVG(e.salary) AS avg_salary
FROM Employees e
JOIN Departments d
ON e.dept_id = d.dept_id
GROUP BY d.dept_name
HAVING AVG(e.salary) >
(
        SELECT AVG(salary)
        FROM Employees
);

8. Clients whose projects exceed 500 total hours

ANSWER
SELECT p.client_name,
              SUM(a.hours_worked) AS total_hours
FROM Projects p
JOIN Assignments a
ON p.project_id = a.project_id
GROUP BY p.client_name
HAVING SUM(a.hours_worked) > 500;

9. Department with highest employee count in each location

ANSWER
SELECT location_id,
              dept_name,
              employee_count
FROM
(
        SELECT d.location_id,
                      d.dept_name,
                      COUNT(e.employee_id) AS employee_count,
                      RANK() OVER(PARTITION BY d.location_id
                      ORDER BY COUNT(e.employee_id) DESC) AS rnk
        FROM Departments d
        LEFT JOIN Employees e
        ON d.dept_id = e.dept_id
        GROUP BY d.location_id, d.dept_name
) x
WHERE rnk = 1;

10. Projects having more than 5 employees assigned

ANSWER
SELECT p.project_name,
              COUNT(DISTINCT a.employee_id) AS total_employees
FROM Projects p
JOIN Assignments a
ON p.project_id = a.project_id
GROUP BY p.project_name
HAVING COUNT(DISTINCT a.employee_id) > 5;

12. Departments having employees assigned to more than one project

ANSWER
SELECT d.dept_name
FROM Departments d
JOIN Employees e
ON d.dept_id = e.dept_id
JOIN Assignments a
ON e.employee_id = a.employee_id
GROUP BY d.dept_name
HAVING COUNT(DISTINCT a.project_id) > 1;

13. Employees who worked less than 10 hours on a project

ANSWER
SELECT e.employee_name,
              p.project_name
FROM Employees e
JOIN Assignments a
ON e.employee_id = a.employee_id
JOIN Projects p
ON a.project_id = p.project_id
WHERE a.hours_worked < 10;

14. Projects with no assigned employees

ANSWER
SELECT p.project_name
FROM Projects p
LEFT JOIN Assignments a
ON p.project_id = a.project_id
WHERE a.assignment_id IS NULL;

16. Employee Name and oldest project start date

ANSWER
SELECT e.employee_name,
              MIN(a.start_date) AS oldest_project_date
FROM Employees e
JOIN Assignments a
ON e.employee_id = a.employee_id
GROUP BY e.employee_name;

17. Projects active in 2023 using DATE_FORMAT

ANSWER
SELECT project_name,
              start_date
FROM Projects
WHERE DATE_FORMAT(start_date, '%Y') = '2023';

18. Employee Names with IFNULL for manager

ANSWER
SELECT e.employee_name,
              IFNULL(m.employee_name, 'CEO') AS manager_name
FROM Employees e
LEFT JOIN Employees m
ON e.manager_id = m.employee_id;

19. Top 3 Departments by total hours worked

ANSWER
SELECT d.dept_name,
              SUM(a.hours_worked) AS total_hours
FROM Departments d
JOIN Employees e
ON d.dept_id = e.dept_id
JOIN Assignments a
ON e.employee_id = a.employee_id
GROUP BY d.dept_name
ORDER BY total_hours DESC
LIMIT 3;

20. Project activity status using CASE

ANSWER
SELECT p.project_name,
              CASE
                      WHEN SUM(a.hours_worked) > 300
                      THEN 'High Activity'
                      ELSE 'Normal'
              END AS status
FROM Projects p
LEFT JOIN Assignments a
ON p.project_id = a.project_id
GROUP BY p.project_name;

21. Employee Name and Manager Name (even if NULL)

ANSWER
SELECT e.employee_name AS employee,
              m.employee_name AS manager
FROM Employees e
LEFT JOIN Employees m
ON e.manager_id = m.employee_id;

23. Employees assigned to same projects as Employee ID 101

ANSWER
SELECT DISTINCT e.employee_name
FROM Employees e
JOIN Assignments a
ON e.employee_id = a.employee_id
WHERE a.project_id IN
(
        SELECT project_id
        FROM Assignments
        WHERE employee_id = 101
)
AND e.employee_id <> 101;
