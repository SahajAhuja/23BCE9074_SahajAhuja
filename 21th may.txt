-- Create Database
CREATE DATABASE company;
USE company;

-- Create Worker Table
CREATE TABLE worker (
    worker_id INT PRIMARY KEY,
    first_name VARCHAR(50),
    department VARCHAR(50),
    salary INT
);

---------------------------------------------------
-- 1. Total Salary Department Wise
---------------------------------------------------

SELECT department, SUM(salary) AS total_salary
FROM worker
GROUP BY department;

---------------------------------------------------
-- 2. Two Departments Having Lowest Total Salaries
---------------------------------------------------

SELECT department, SUM(salary) AS total_salary
FROM worker
GROUP BY department
ORDER BY total_salary ASC
LIMIT 2;

---------------------------------------------------
-- 3. Average Salary Department Wise
---------------------------------------------------

SELECT department, AVG(salary) AS avg_salary
FROM worker
GROUP BY department;

---------------------------------------------------
-- 4. Two Departments Having Lowest Average Salaries
---------------------------------------------------

SELECT department, AVG(salary) AS avg_salary
FROM worker
GROUP BY department
ORDER BY avg_salary ASC
LIMIT 2;

---------------------------------------------------
-- 5. Highest Salary Employee From Each Department
---------------------------------------------------

SELECT first_name, department, salary
FROM worker w
WHERE salary = (
    SELECT MAX(salary)
    FROM worker
    WHERE department = w.department
);

---------------------------------------------------
-- 6. Department Wise Maximum Salary
---------------------------------------------------

SELECT department, MAX(salary) AS max_salary
FROM worker
GROUP BY department;

---------------------------------------------------
-- 7. Department Wise Minimum Salary
---------------------------------------------------

SELECT department, MIN(salary) AS min_salary
FROM worker
GROUP BY department;

---------------------------------------------------
-- 8. Department Wise Employee Count
---------------------------------------------------

SELECT department, COUNT(*) AS employee_count
FROM worker
GROUP BY department;
